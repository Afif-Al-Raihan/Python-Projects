import random

secret_code = random.randint(1, 100)

def easy():
    i = 10
    while i>0:
        user = int(input("Enter a number: "))
        if user==secret_code:
            print("You won!")
            break
        elif user!=secret_code and user < secret_code:
            print("Too Low! Try again")
            print("remaining tries: ", i)
        elif user!=secret_code and user>secret_code:
            print("Too High! Try again")
            print("remaining tries: ", i)
        else:
            print("Game over")
            break
        i-=1
