import pygame, sys
import random

# Initialize Pygame
pygame.init()

# Constants
WIDTH, HEIGHT = 900, 700
pipe_width = 70
pipe_height = 200
pipe_color = (0, 128, 0)
bg_color = (135, 206, 235)
gravity = 1
jump = -14

# Create the game window
screen = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption("Flappy Bird")



# Bird initial position
bird_x = 50
bird_y = HEIGHT // 2
bird_velocity = 0

# Load the bird image
bird = pygame.image.load("bird.png")  # Replace "bird.png" with your bird image file path
bird_rect = bird.get_rect()
bird_rect.center = (bird_x, bird_y)


bird_radius = 24
# Pipe initial position
pipe_x = WIDTH
pipe_y = random.randint(100, 400)
pipe_speed = 10

# Score
score = 0

# Fonts
font = pygame.font.Font(None, 36)

# Game over flag
game_over = False

# Game loop
while True:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            pygame.quit()
            sys.exit()
        if event.type == pygame.KEYDOWN and not game_over:
            if event.key == pygame.K_SPACE:
                bird_velocity = jump

    if not game_over:
        # Update bird position
        bird_y += bird_velocity
        bird_velocity += gravity

        # Update pipe position
        pipe_x -= pipe_speed

        # Check for collisions
        if bird_y > HEIGHT or bird_y < 0:
            game_over = True
        if pipe_x < -pipe_width:
            pipe_x = WIDTH
            pipe_y = random.randint(100, 400)
        if bird_x + bird_radius > pipe_x and bird_x - bird_radius < pipe_x + pipe_width:
            if bird_y - bird_radius < pipe_y or bird_y + bird_radius > pipe_y + pipe_height:
                game_over = True
        if bird_x == pipe_x + pipe_width:
            score += 1

        # Clear the screen
        screen.fill(bg_color)

        # Draw the bird
        bird_rect.center = (bird_x, bird_y)

        # Draw the bird image
        screen.blit(bird, bird_rect)

        # Draw the pipe
        pygame.draw.rect(screen, pipe_color, (pipe_x, 0, pipe_width, pipe_y))
        pygame.draw.rect(screen, pipe_color, (pipe_x, pipe_y + pipe_height, pipe_width, HEIGHT))

        # Display score
        text = font.render("Score: " + str(score), True, (0, 0, 0))
        screen.blit(text, (10, 10))

        # Update the display
        pygame.display.flip()

        # Limit the frame rate
        pygame.time.delay(30)
    else:
        # Game over screen
        font = pygame.font.Font(None, 72)
        text = font.render("Game Over", True, (255, 0, 0))
        screen.blit(text, (WIDTH // 2 - 150, HEIGHT // 2 - 50))
        font = pygame.font.Font(None, 36)
        text = font.render("Press 'R' to restart", True, (0, 0, 0))
        screen.blit(text, (WIDTH // 2 - 100, HEIGHT // 2 + 50))
        pygame.display.flip()

    # Check for restart
    keys = pygame.key.get_pressed()
    if keys[pygame.K_r]:
        game_over = False
        bird_x = 50
        bird_y = HEIGHT // 2
        bird_velocity = 0
        pipe_x = WIDTH
        pipe_y = random.randint(100, 400)
        score = 0
