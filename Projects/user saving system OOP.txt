class Human:
    def __init__(self, name, age, blood_type):
        self.name = name
        self.age = age
        self.blood_type = blood_type

name = input("Enter your Name: ")
age = int(input("Enter your Date Of Birth (DOB): "))
BloodType = input("Enter your Blood Type: ")

date_of_birth = 2023 - age

with open('save.txt', 'a') as f:
    p1 = Human(name, age, BloodType)
    f.write("Name: "+ p1.name)
    f.write('\n')
    f.write("Age: "+ str(date_of_birth))
    f.write('\n')
    f.write("Blood Type: "+ BloodType)

print("Human form has been filled up")
print("Here is your data\n")

with open('save.txt', 'r') as f:
    contents = f.read()
    print(contents)
